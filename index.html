// (or whatever element displays this) const delayRange = document.getElementById("delayRange"); const rangeBullet = document.getElementById("rs-bullet"); const delayMicrosecondsInput = document.getElementById("delayMicroseconds"); // Make sure this element exists function updateUI() { const value = parseInt(delayRange.value, 10); const min = parseInt(delayRange.min, 10); // Get min from the input const max = parseInt(delayRange.max, 10); // Get max from the input const seconds = (value * 0.5).toFixed(1); const microseconds = value * 500000; rangeBullet.textContent = seconds; if (delayMicrosecondsInput) { // Check if the element exists before trying to update it delayMicrosecondsInput.value = microseconds; } // --- Core logic for precise circle positioning --- const sliderWidth = delayRange.offsetWidth; // Total width of the input element const thumbWidth = 22; // From your CSS: custom-range::-webkit-slider-thumb width const percentage = (value - min) / (max - min); // Current value as a percentage of the total range // Calculate the effective width the thumb can move within (track width) const trackWidth = sliderWidth - thumbWidth; // Calculate the horizontal position of the center of the thumb // This position is relative to the left edge of the track const thumbCenterPosition = (percentage * trackWidth) + (thumbWidth / 2); // Apply the calculated position to the rangeBullet (circle label) // The `transform: translateX(-50%)` in your CSS will then perfectly // center the label on this `left` point. rangeBullet.style.left = `${thumbCenterPosition}px`; // --- End core logic --- } // Attach event listener for real-time updates delayRange.addEventListener("input", updateUI); // Call updateUI once on load to set initial position and values window.addEventListener("load", updateUI); // Also call updateUI if DOMContentLoaded fires later or element is ready // This helps if the script is placed early in the HTML updateUI(); });
